    "### 4.5.2　逐次特徴選択アルゴリズム\n",
    "\n",
    "- 次元削減\n",
    "- 特徴選択\n",
    "\n",
    "  - 特徴量の一部を選択する\n",
    "  - 計算効率を改善する\n",
    "  - 無関係な特徴量を取り除くことで汎化性能を向上させる\n",
    "  - 正則化を使えないアルゴリズムで役立つ\n",
    "  \n",
    "- 特徴抽出\n",
    "\n",
    "  - 特徴空間からより次元の低い特徴空間へマッピングする\n",
    "  \n",
    "- 逐次特徴選択\n",
    "\n",
    "  - 貪欲探索\n",
    "  \n",
    "- 逐次後退選択 (SBS)\n",
    "\n",
    "  - 目的の個数の特徴量まで特徴量を逐次的に削除する\n",
    "  - $J$: 最小化問題の目的関数\n",
    "  - $d$: 元の特徴量数\n",
    "  - $X_k$: $k$ 次元特徴空間\n",
    "  \n",
    "  1. $k = d$ でアルゴリズムを初期化\n",
    "  2. 最小化問題において $J$ を最大化する特徴量 $x^-$ を決定する: $x^- = argmax J(X_k - x)$\n",
    "  3. $x^-$ を削除する: $X_{k_1} = X_k - x^-; k := k - 1$"
   ]

    "from sklearn.base import clone\n",
    "from itertools import combinations\n",
    "import numpy\n",
    "from sklearn.metrics import accuracy_score\n",
    "\n",
    "\n",
    "class SBS():\n",
    "    \"\"\" Sequential backward selection \"\"\"\n",
    "\n",
    "    def __init__(self, estimator, k_features, scoring=accuracy_score,\n",
    "                 test_size=0.25, random_state=1):\n",
    "        self.scoring = scoring\n",
    "        self.estimator = clone(estimator)\n",
    "        self.k_features = k_features\n",
    "        self.test_size = test_size\n",
    "        self.random_state = random_state\n",
    "        self.indices_ = None\n",
    "        self.subsets_ = None\n",
    "\n",
    "    def fit(self, X, y):\n",
    "        X_train, X_test, y_train, y_test = train_test_split(\n",
    "                X, y,\n",
    "                test_size=self.test_size,\n",
    "                random_state=self.random_state,\n",
    "        )\n",
    "        dim = X_train.shape[1]\n",
    "        self.indices_ = tuple(range(dim))\n",
    "        self.subsets_ = [self.indices_]\n",
    "        score = self._calc_score(X_train, y_train,\n",
    "                X_test, y_test, self.indices_)\n",
    "        self.scores_ = [score]\n",
    "        while dim > self.k_features:\n",
    "            scores = []\n",
    "            subsets = []\n",
    "            for p in combinations(self.indices_, r=dim - 1):\n",
    "                score = self._calc_score(X_train, y_train, X_test, y_test, p)\n",
    "                scores.append(score)\n",
    "                subsets.append(p)\n",
    "                \n",
    "            best = numpy.argmax(scores)\n",
    "            self.indices_ = subsets[best]\n",
    "            self.subsets_.append(self.indices_)\n",
    "            dim -= 1\n",
    "            self.scores_.append(scores[best])\n",
    "        \n",
    "        self.k_score_ = self.scores_[-1]    \n",
    "        return self\n",
    "\n",
    "    def transform(self, X):\n",
    "        return X[:, self.indices_]\n",
    "\n",
    "    def _calc_score(self, X_train, y_train, X_test, y_test, indices):\n",
    "        self.estimator.fit(X_train[:, indices], y_train)\n",
    "        y_pred = self.estimator.predict(X_test[:, indices])\n",
    "        score = self.scoring(y_test, y_pred)\n",
    "        return score\n"
